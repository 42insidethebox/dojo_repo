int	main(int argc, char **argv)
{
	if (argc < 4)
		print_usage();

	// Split argv into command groups
	char	***cmds = parse_commands(argc, argv);
	if (!cmds)
	{
		fprintf(stderr, "Failed to parse commands\n");
		return (1);
	}

	execute_pipeline(cmds);

	free_commands(cmds);
	return (0);
}
